		MAX(CASE WHEN a.dias = 'Martes' AND a.turno = 'Noche' THEN a.id_agenda ELSE NULL END) as id_mar_n,
		
   
    -- MIÉRCOLES

    MAX(CASE WHEN a.dias = 'Miércoles' AND a.turno = 'Tarde' AND a.estado = 'Abierto' THEN 2 WHEN a.dias = 'Miércoles' AND a.turno = 'Tarde' AND a.estado = 'Cerrado' THEN 1 ELSE 0 END) as mie_t,
		MAX(CASE WHEN a.dias = 'Miércoles' AND a.turno = 'Tarde' THEN a.id_agenda ELSE NULL END) as id_mie_t,
		
    MAX(CASE WHEN a.dias = 'Miércoles' AND a.turno = 'Noche' AND a.estado = 'Abierto' THEN 2 WHEN a.dias = 'Miércoles' AND a.turno = 'Noche' AND a.estado = 'Cerrado' THEN 1 ELSE 0 END) as mie_n,

    MAX(CASE WHEN a.dias = 'Jueves' AND a.turno = 'Tarde' AND a.estado = 'Abierto' THEN 2 WHEN a.dias = 'Jueves' AND a.turno = 'Tarde' AND a.estado = 'Cerrado' THEN 1 ELSE 0 END) as jue_t,
		MAX(CASE WHEN a.dias = 'Jueves' AND a.turno = 'Tarde' THEN a.id_agenda ELSE NULL END) as id_jue_t,
		
    MAX(CASE WHEN a.dias = 'Jueves' AND a.turno = 'Noche' AND a.estado = 'Abierto' THEN 2 WHEN a.dias = 'Jueves' AND a.turno = 'Noche' AND a.estado = 'Cerrado' THEN 1 ELSE 0 END) as jue_n,

		MAX(CASE WHEN a.dias = 'Domingo' AND a.turno = 'Noche' THEN a.id_agenda ELSE NULL END) as id_dom_n
    
FROM maquinas m
LEFT JOIN agendas a ON m.id_maquina = a.id_maquina
LEFT JOIN modalidades md ON m.id_modalidad = md.id_modalidad
 -- Filtro modalidades
WHERE 
    ( {{ !SelectModalidad.selectedOptionValue }} OR m.id_modalidad = {{ SelectModalidad.selectedOptionValue }} )
		
GROUP BY m.id_maquina, m.nombre_maquina, md.nombre_modalidad;
